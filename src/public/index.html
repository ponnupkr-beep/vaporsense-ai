import express from 'express'; import dotenv from 'dotenv'; import path from 'path'; dotenv.config(); const app = express(); const PORT = process.env.PORT || 10000; app.use(express.json()); // Serve welcome page app.use(express.static(path.join(__dirname, '../src/public'))); // Health check endpoint app.get('/api/health', (req, res) => { res.json({ status: 'ok', message: 'VaporSense AI is running! 🚀' }); }); app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); }); export default app;
